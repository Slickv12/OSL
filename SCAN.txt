#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, total = 0, i, j, disk_size, direction;
    int requests[100], temp;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for (i = 0; i < n; i++)
        scanf("%d", &requests[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter total disk size: ");
    scanf("%d", &disk_size);

    printf("Enter head movement direction (0 for left, 1 for right): ");
    scanf("%d", &direction);

    // Sort requests
    for (i = 0; i < n - 1; i++)
        for (j = 0; j < n - i - 1; j++)
            if (requests[j] > requests[j + 1]) {
                temp = requests[j];
                requests[j] = requests[j + 1];
                requests[j + 1] = temp;
            }

    int pos = 0;
    for (i = 0; i < n; i++) {
        if (head < requests[i]) {
            pos = i;
            break;
        }
    }

    if (direction == 1) {  // moving right
        for (i = pos; i < n; i++) {
            total += abs(requests[i] - head);
            head = requests[i];
        }
        total += abs((disk_size - 1) - head);
        head = disk_size - 1;
        for (i = pos - 1; i >= 0; i--) {
            total += abs(requests[i] - head);
            head = requests[i];
        }
    } else {  // moving left
        for (i = pos - 1; i >= 0; i--) {
            total += abs(requests[i] - head);
            head = requests[i];
        }
        total += abs(head - 0);
        head = 0;
        for (i = pos; i < n; i++) {
            total += abs(requests[i] - head);
            head = requests[i];
        }
    }

    printf("\nTotal Seek Time: %d\n", total);
    printf("Average Seek Time: %.2f\n", (float)total / n);

    return 0;
}

#include <stdio.h>

int main() {
    int frames[10], pages[30];
    int n, m, pageFaults = 0, pageHits = 0, flag, i, j, k = 0;

    printf("Enter number of pages: ");
    scanf("%d", &m);

    printf("Enter number of frames: ");
    scanf("%d", &n);

    printf("Enter the page reference string:\n");
    for (i = 0; i < m; i++)
        scanf("%d", &pages[i]);

    for (i = 0; i < n; i++)
        frames[i] = -1;

    for (i = 0; i < m; i++) {
        flag = 0;
        for (j = 0; j < n; j++) {
            if (frames[j] == pages[i]) {
                flag = 1;
                pageHits++;
                break;
            }
        }

        if (flag == 0) {
            frames[k] = pages[i];
            k = (k + 1) % n;
            pageFaults++;
        }
    }

    printf("\nTotal Page Faults: %d\n", pageFaults);
    printf("Total Page Hits: %d\n", pageHits);

    return 0;
}

#include <stdio.h>

int main() {
    int frames[10], pages[30];
    int n, m, pageFaults = 0, pageHits = 0;
    int i, j, k, pos, flag1, flag2, max, temp[10];

    printf("Enter number of pages: ");
    scanf("%d", &m);

    printf("Enter number of frames: ");
    scanf("%d", &n);

    printf("Enter the page reference string:\n");
    for (i = 0; i < m; i++)
        scanf("%d", &pages[i]);

    for (i = 0; i < n; i++)
        frames[i] = -1;

    for (i = 0; i < m; i++) {
        flag1 = flag2 = 0;

        for (j = 0; j < n; j++) {
            if (frames[j] == pages[i]) {
                flag1 = flag2 = 1;
                pageHits++;
                break;
            }
        }

        if (flag1 == 0) {
            for (j = 0; j < n; j++) {
                if (frames[j] == -1) {
                    frames[j] = pages[i];
                    flag2 = 1;
                    pageFaults++;
                    break;
                }
            }
        }

        if (flag2 == 0) {
            for (j = 0; j < n; j++) {
                temp[j] = -1;
                for (k = i + 1; k < m; k++) {
                    if (frames[j] == pages[k]) {
                        temp[j] = k;
                        break;
                    }
                }
            }

            pos = -1;
            max = -1;
            for (j = 0; j < n; j++) {
                if (temp[j] == -1) {
                    pos = j;
                    break;
                } else if (temp[j] > max) {
                    max = temp[j];
                    pos = j;
                }
            }

            frames[pos] = pages[i];
            pageFaults++;
        }
    }

    printf("\nTotal Page Faults: %d\n", pageFaults);
    printf("Total Page Hits: %d\n", pageHits);

    return 0;
}
